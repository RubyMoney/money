def lookup(key, _)
  i18n_key = KEY_MAP[key]

  ::I18n.t i18n_key, scope: 'number.currency.format', raise: true
rescue ::I18n::MissingTranslationData
  ::I18n.t i18n_key, scope: 'number.format', default: nil
end
